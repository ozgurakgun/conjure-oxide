9/// A Minion model.
10#[non_exhaustive]
11#[derive(Debug, Clone, PartialEq, Eq)]
12pub struct Model {
13    pub named_variables: SymbolTable,
14    pub constraints: Vec<Constraint>,
15}

20        Model {
21            named_variables: SymbolTable::new(),
22            constraints: Vec::new(),
23        }
24    }
25}

28    fn default() -> Self {

118/// Representation of a Minion Variable.
119///
120/// A variable can either be a named variable, or an anomynous "constant as a variable".
121///
122/// The latter is not stored in the symbol table, or counted in Minions internal list of all
123/// variables, but is used to allow the use of a constant in the place of a variable in a
124/// constraint.
125#[derive(Debug, Clone, Eq, PartialEq)]
126pub enum Var {
127    NameRef(VarName),
128    ConstantAsVar(i32),
129}
130
131/// Representation of a Minion constant.
132#[non_exhaustive]
133#[derive(Debug, Eq, PartialEq, Clone, Copy)]
134pub enum Constant {
135    Bool(bool),
136    Integer(i32),
137}

139/// Representation of variable domains.
140#[derive(Debug, Copy, Clone, Eq, PartialEq)]
141#[non_exhaustive]
142pub enum VarDomain {
143    Bound(i32, i32),
144    Discrete(i32, i32),
145    SparseBound(i32, i32),
146    Bool,
147}
148
149#[derive(Debug, Clone, Eq, PartialEq)]
150#[non_exhaustive]
151/// Stores all named variables in a Minion model alongside their domains.
152///
153/// Named variables referenced in [constraints](Constraint) must be in the symbol table for the
154/// model to be valid. In the future, this will raise some sort of type error.
155pub struct SymbolTable {
156    table: HashMap<VarName, VarDomain>,
157
158    // order of all variables
159    var_order: Vec<VarName>,
160
161    // search order
162    search_var_order: Vec<VarName>,
163}
164
165impl SymbolTable {
166    fn new() -> SymbolTable {
167        SymbolTable {
168            table: HashMap::new(),
169            var_order: Vec::new(),
170            search_var_order: Vec::new(),
171        }
172    }
173
174    /// Creates a new search variable and adds it to the symbol table.
175    ///
176    /// # Returns
177    ///
178    /// If a variable already exists with the given name, `None` is returned.
179    pub fn add_var(&mut self, name: VarName, vartype: VarDomain) -> Option<()> {
180        if self.table.contains_key(&name) {
181            return None;
182        }
183
184        self.table.insert(name.clone(), vartype);
185        self.var_order.push(name.clone());
186        self.search_var_order.push(name);
187
188        Some(())
189    }
190
191    /// Creates a new auxiliary variable and adds it to the symbol table.
192    ///
193    /// This variable will excluded from Minions search and printing order.
194    ///
195    /// # Returns
196    ///
197    /// If a variable already exists with the given name, `None` is returned.
198    pub fn add_aux_var(&mut self, name: VarName, vartype: VarDomain) -> Option<()> {
199        if self.table.contains_key(&name) {
200            return None;
201        }
202
203        self.table.insert(name.clone(), vartype);
204        self.var_order.push(name);
205
206        Some(())
207    }
208
209    /// Gets the domain of a named variable.
210    ///
211    /// # Returns
212    ///
213    /// `None` if no variable is known by that name.
214    pub fn get_vartype(&self, name: VarName) -> Option<VarDomain> {
215        self.table.get(&name).cloned()
216    }
217
218    /// Gets the canonical ordering of all variables.
219    pub fn get_variable_order(&self) -> Vec<VarName> {
220        self.var_order.clone()
221    }
222
223    /// Gets the canonical ordering of search variables.
224    pub fn get_search_variable_order(&self) -> Vec<VarName> {
225        self.search_var_order.clone()
226    }
227
228    pub fn contains(&self, name: VarName) -> bool {
229        self.table.contains_key(&name)
230    }
