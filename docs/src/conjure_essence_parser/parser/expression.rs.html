12/// Parse an Essence expression into its Conjure AST representation.
13pub fn parse_expression(
14    constraint: Node,
15    source_code: &str,
16    root: &Node,
17) -> Result<Expression, EssenceParseError> {
18    // TODO (gskorokhod) - Factor this further (make match arms into separate functions, extract common logic)
19    match constraint.kind() {
20        "constraint" | "expression" | "boolean_expr" | "comparison_expr" | "arithmetic_expr"
21        | "primary_expr" | "sub_expr" => child_expr(constraint, source_code, root),
22        "not_expr" => Ok(Expression::Not(
23            Metadata::new(),
24            Box::new(child_expr(constraint, source_code, root)?),
25        )),
26        "abs_value" => Ok(Expression::Abs(
27            Metadata::new(),
28            Box::new(child_expr(constraint, source_code, root)?),
29        )),
30        "negative_expr" => Ok(Expression::Neg(
31            Metadata::new(),
32            Box::new(child_expr(constraint, source_code, root)?),
33        )),
34        "exponent" | "product_expr" | "sum_expr" | "comparison" | "and_expr" | "or_expr"
35        | "implication" => {
36            let expr1 = child_expr(constraint, source_code, root)?;
37            let op = constraint.child(1).ok_or(format!(
38                "Missing operator in expression {}",
39                constraint.kind()
40            ))?;
41            let op_type = &source_code[op.start_byte()..op.end_byte()];
42            let expr2_node = constraint.child(2).ok_or(format!(
43                "Missing second operand in expression {}",
44                constraint.kind()
45            ))?;
46            let expr2 = parse_expression(expr2_node, source_code, root)?;
47
48            match op_type {
49                "**" => Ok(Expression::UnsafePow(
50                    Metadata::new(),
51                    Box::new(expr1),
52                    Box::new(expr2),
53                )),
54                "+" => Ok(Expression::Sum(
55                    Metadata::new(),
56                    Box::new(matrix_expr![expr1, expr2]),
57                )),
58                "-" => Ok(Expression::Minus(
59                    Metadata::new(),
60                    Box::new(expr1),
61                    Box::new(expr2),
62                )),
63                "*" => Ok(Expression::Product(Metadata::new(), vec![expr1, expr2])),
64                "/" => {
65                    //TODO: add checks for if division is safe or not
66                    Ok(Expression::UnsafeDiv(
67                        Metadata::new(),
68                        Box::new(expr1),
69                        Box::new(expr2),
70                    ))
71                }
72                "%" => {
73                    //TODO: add checks for if mod is safe or not
74                    Ok(Expression::UnsafeMod(
75                        Metadata::new(),
76                        Box::new(expr1),
77                        Box::new(expr2),
78                    ))
79                }
80                "=" => Ok(Expression::Eq(
81                    Metadata::new(),
82                    Box::new(expr1),
83                    Box::new(expr2),
84                )),
85                "!=" => Ok(Expression::Neq(
86                    Metadata::new(),
87                    Box::new(expr1),
88                    Box::new(expr2),
89                )),
90                "<=" => Ok(Expression::Leq(
91                    Metadata::new(),
92                    Box::new(expr1),
93                    Box::new(expr2),
94                )),
95                ">=" => Ok(Expression::Geq(
96                    Metadata::new(),
97                    Box::new(expr1),
98                    Box::new(expr2),
99                )),
100                "<" => Ok(Expression::Lt(
101                    Metadata::new(),
102                    Box::new(expr1),
103                    Box::new(expr2),
104                )),
105                ">" => Ok(Expression::Gt(
106                    Metadata::new(),
107                    Box::new(expr1),
108                    Box::new(expr2),
109                )),
110                "/\\" => Ok(Expression::And(
111                    Metadata::new(),
112                    Box::new(matrix_expr![expr1, expr2]),
113                )),
114                "\\/" => Ok(Expression::Or(
115                    Metadata::new(),
116                    Box::new(matrix_expr![expr1, expr2]),
117                )),
118                "->" => Ok(Expression::Imply(
119                    Metadata::new(),
120                    Box::new(expr1),
121                    Box::new(expr2),
122                )),
123                _ => Err(format!("Unsupported operator '{}'", op_type).into()),
124            }
125        }
126        "quantifier_expr" => {
127            let mut expr_list = Vec::new();
128            for expr in named_children(&constraint) {
129                expr_list.push(parse_expression(expr, source_code, root)?);
130            }
131            let quantifier = constraint.child(0).ok_or(format!(
132                "Missing quantifier in expression {}",
133                constraint.kind()
134            ))?;
135            let quantifier_type = &source_code[quantifier.start_byte()..quantifier.end_byte()];
136
137            match quantifier_type {
138                "and" => Ok(Expression::And(
139                    Metadata::new(),
140                    Box::new(into_matrix_expr![expr_list]),
141                )),
142                "or" => Ok(Expression::Or(
143                    Metadata::new(),
144                    Box::new(into_matrix_expr![expr_list]),
145                )),
146                "min" => Ok(Expression::Min(
147                    Metadata::new(),
148                    Box::new(into_matrix_expr![expr_list]),
149                )),
150                "max" => Ok(Expression::Max(
151                    Metadata::new(),
152                    Box::new(into_matrix_expr![expr_list]),
153                )),
154                "sum" => Ok(Expression::Sum(
155                    Metadata::new(),
156                    Box::new(into_matrix_expr![expr_list]),
157                )),
158                "allDiff" => Ok(Expression::AllDiff(
159                    Metadata::new(),
160                    Box::new(into_matrix_expr![expr_list]),
161                )),
162                _ => Err(format!("Unsupported quantifier {}", constraint.kind()).into()),
163            }
164        }
165        "constant" => {
166            let child = constraint.child(0).ok_or(format!(
167                "Missing value for constant expression {}",
168                constraint.kind()
169            ))?;
170            match child.kind() {
171                "integer" => {
172                    let constant_value = &source_code[child.start_byte()..child.end_byte()]
173                        .parse::<i32>()
174                        .unwrap();
175                    Ok(Expression::Atomic(
176                        Metadata::new(),
177                        Atom::Literal(Literal::Int(*constant_value)),
178                    ))
179                }
180                "TRUE" => Ok(Expression::Atomic(
181                    Metadata::new(),
182                    Atom::Literal(Literal::Bool(true)),
183                )),
184                "FALSE" => Ok(Expression::Atomic(
185                    Metadata::new(),
186                    Atom::Literal(Literal::Bool(false)),
187                )),
188                _ => Err(format!("Unsupported constant kind: {}", child.kind()).into()),
189            }
190        }
191        "variable" => {
192            let variable_name =
193                String::from(&source_code[constraint.start_byte()..constraint.end_byte()]);
194            Ok(Expression::Atomic(
195                Metadata::new(),
196                Atom::Reference(Name::UserName(variable_name)),
197            ))
198        }
199        "from_solution" => match root.kind() {
200            "dominance_relation" => {
201                let inner = child_expr(constraint, source_code, root)?;
202                match inner {
203                    Expression::Atomic(_, _) => {
204                        Ok(Expression::FromSolution(Metadata::new(), Box::new(inner)))
205                    }
206                    _ => Err(
207                        "Expression inside a `fromSolution()` must be a variable name"
208                            .to_string()
209                            .into(),
210                    ),
211                }
212            }
213            _ => Err(
214                "`fromSolution()` is only allowed inside dominance relation definitions"
215                    .to_string()
216                    .into(),
217            ),
218        },
219        _ => Err(format!("{} is not a recognized node kind", constraint.kind()).into()),
220    }
221}

223pub fn child_expr(
224    node: Node,
225    source_code: &str,
226    root: &Node,
227) -> Result<Expression, EssenceParseError> {
228    match node.named_child(0) {
229        Some(child) => parse_expression(child, source_code, root),
230        None => Err(format!("Missing node in expression of kind {}", node.kind()).into()),
231    }
232}
