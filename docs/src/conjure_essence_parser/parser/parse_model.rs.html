21/// Parse an Essence file into a Model using the tree-sitter parser.
22pub fn parse_essence_file_native(
23    path: &str,
24    context: Arc<RwLock<Context<'static>>>,
25) -> Result<Model, EssenceParseError> {
26    let source_code = fs::read_to_string(path)
27        .unwrap_or_else(|_| panic!("Failed to read the source code file {}", path));
28    parse_essence_with_context(&source_code, context)
29}

31pub fn parse_essence_with_context(
32    src: &str,
33    context: Arc<RwLock<Context<'static>>>,
34) -> Result<Model, EssenceParseError> {
35    let (tree, source_code) = match get_tree(src) {
36        Some(tree) => tree,
37        None => {
38            return Err(EssenceParseError::TreeSitterError(
39                "Failed to parse source code".to_string(),
40            ))
41        }
42    };
43
44    let mut model = Model::new(context);
45    let root_node = tree.root_node();
46    for statement in named_children(&root_node) {
47        match statement.kind() {
48            "single_line_comment" => {}
49            "find_statement_list" => {
50                let var_hashmap = parse_find_statement(statement, &source_code);
51                for (name, decision_variable) in var_hashmap {
52                    model
53                        .as_submodel_mut()
54                        .symbols_mut()
55                        .insert(Rc::new(Declaration::new_var(name, decision_variable)));
56                }
57            }
58            "constraint_list" => {
59                let mut constraint_vec: Vec<Expression> = Vec::new();
60                for constraint in named_children(&statement) {
61                    if constraint.kind() != "single_line_comment" {
62                        constraint_vec.push(parse_expression(
63                            constraint,
64                            &source_code,
65                            &statement,
66                        )?);
67                    }
68                }
69                model.as_submodel_mut().add_constraints(constraint_vec);
70            }
71            "language_label" => {}
72            "letting_statement_list" => {
73                let letting_vars = parse_letting_statement(statement, &source_code)?;
74                model.as_submodel_mut().symbols_mut().extend(letting_vars);
75            }
76            "dominance_relation" => {
77                let inner = statement
78                    .child(1)
79                    .expect("Expected a sub-expression inside `dominanceRelation`");
80                let expr = parse_expression(inner, &source_code, &statement)?;
81                let dominance = Expression::DominanceRelation(Metadata::new(), Box::new(expr));
82                if model.dominance.is_some() {
83                    return Err(EssenceParseError::ParseError(Error::Parse(
84                        "Duplicate dominance relation".to_owned(),
85                    )));
86                }
87                model.dominance = Some(dominance);
88            }
89            _ => {
90                let kind = statement.kind();
91                return Err(EssenceParseError::ParseError(Error::Parse(
92                    format!("Unrecognized top level statement kind: {kind}").to_owned(),
93                )));
94            }
95        }
96    }
97    Ok(model)
98}
