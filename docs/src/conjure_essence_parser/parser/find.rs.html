11/// Parse a find statement into a map of decision variable names to their domains.
12pub fn parse_find_statement(
13    find_statement_list: Node,
14    source_code: &str,
15) -> BTreeMap<Name, Domain> {
16    let mut vars = BTreeMap::new();
17
18    for find_statement in named_children(&find_statement_list) {
19        let mut temp_symbols = BTreeSet::new();
20
21        let variable_list = find_statement
22            .named_child(0)
23            .expect("No variable list found");
24        for variable in named_children(&variable_list) {
25            let variable_name = &source_code[variable.start_byte()..variable.end_byte()];
26            temp_symbols.insert(variable_name);
27        }
28
29        let domain = find_statement.named_child(1).expect("No domain found");
30        let domain = parse_domain(domain, source_code);
31
32        for name in temp_symbols {
33            vars.insert(Name::UserName(String::from(name)), domain.clone());
34        }
35    }
36    vars
37}
