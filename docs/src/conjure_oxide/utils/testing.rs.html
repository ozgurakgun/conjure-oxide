26pub fn assert_eq_any_order<T: Eq + Hash + Debug + Clone>(a: &Vec<Vec<T>>, b: &Vec<Vec<T>>) {
27    assert_eq!(a.len(), b.len());
28
29    let mut a_rows: Vec<HashSet<T>> = Vec::new();
30    for row in a {
31        let hash_row = to_set(row);
32        a_rows.push(hash_row);
33    }
34
35    let mut b_rows: Vec<HashSet<T>> = Vec::new();
36    for row in b {
37        let hash_row = to_set(row);
38        b_rows.push(hash_row);
39    }
40
41    println!("{:?},{:?}", a_rows, b_rows);
42    for row in a_rows {
43        assert!(b_rows.contains(&row));
44    }
45}

47pub fn serialise_model(model: &ConjureModel) -> Result<String, JsonError> {
48    // A consistent sorting of the keys of json objects
49    // only required for the generated version
50    // since the expected version will already be sorted
51    let serde_model: SerdeModel = model.clone().into();
52    let generated_json = sort_json_object(&serde_json::to_value(serde_model)?, false);
53
54    // serialise to string
55    let generated_json_str = serde_json::to_string_pretty(&generated_json)?;
56
57    Ok(generated_json_str)
58}

106pub fn minion_solutions_from_json(
107    serialized: &str,
108) -> Result<Vec<HashMap<Name, Literal>>, anyhow::Error> {
109    let json: JsonValue = serde_json::from_str(serialized)?;
110
111    let json_array = json
112        .as_array()
113        .ok_or(Error::Parse("Invalid JSON".to_owned()))?;
114
115    let mut solutions = Vec::new();
116
117    for solution in json_array {
118        let mut sol = HashMap::new();
119        let solution = solution
120            .as_object()
121            .ok_or(Error::Parse("Invalid JSON".to_owned()))?;
122
123        for (var_name, constant) in solution {
124            let constant = match constant {
125                JsonValue::Number(n) => {
126                    let n = n
127                        .as_i64()
128                        .ok_or(Error::Parse("Invalid integer".to_owned()))?;
129                    Literal::Int(n as i32)
130                }
131                JsonValue::Bool(b) => Literal::Bool(*b),
132                _ => return Err(Error::Parse("Invalid constant".to_owned()).into()),
133            };
134
135            sol.insert(UserName(var_name.into()), constant);
136        }
137
138        solutions.push(sol);
139    }
140
141    Ok(solutions)
142}

186/// Reads a rule trace from a file. For the generated prefix, it appends a count message.
187/// Returns the lines of the file as a vector of strings.
188pub fn read_rule_trace(
189    path: &str,
190    test_name: &str,
191    prefix: &str,
192) -> Result<Vec<String>, std::io::Error> {
193    let filename = format!("{path}/{test_name}-{prefix}-rule-trace.json");
194    let mut rules_trace: Vec<String> = read_to_string(&filename)?
195        .lines()
196        .map(String::from)
197        .collect();
198
199    // If prefix is "generated", append the count message
200    if prefix == "generated" {
201        let rule_count = rules_trace.len();
202        let count_message = json!({
203            "message": "Number of rules applied",
204            "count": rule_count
205        });
206        let count_message_string = serde_json::to_string(&count_message)?;
207        rules_trace.push(count_message_string);
208
209        // Overwrite the file with updated content (including the count message)
210        let mut file = OpenOptions::new()
211            .write(true)
212            .truncate(true)
213            .open(&filename)?;
214        writeln!(file, "{}", rules_trace.join("\n"))?;
215    }
216
217    Ok(rules_trace)
218}

236pub fn normalize_solutions_for_comparison(
237    input_solutions: &[BTreeMap<Name, Literal>],
238) -> Vec<BTreeMap<Name, Literal>> {
239    let mut normalized = input_solutions.to_vec();
240
241    for solset in &mut normalized {
242        // remove machine names
243        let keys_to_remove: Vec<Name> = solset
244            .keys()
245            .filter(|k| matches!(k, Name::MachineName(_)))
246            .cloned()
247            .collect();
248        for k in keys_to_remove {
249            solset.remove(&k);
250        }
251
252        let mut updates = vec![];
253        for (k, v) in solset.clone() {
254            if let Name::UserName(_) = k {
255                match v {
256                    Literal::Bool(true) => updates.push((k, Literal::Int(1))),
257                    Literal::Bool(false) => updates.push((k, Literal::Int(0))),
258                    Literal::AbstractLiteral(AbstractLiteral::Matrix(elems, _)) => {
259                        // make all domains the same (this is just in the tester so the types dont
260                        // actually matter)
261
262                        let mut matrix = AbstractLiteral::Matrix(elems, Domain::IntDomain(vec![]));
263                        matrix =
264                            matrix.transform(Arc::new(
265                                move |x: AbstractLiteral<Literal>| match x {
266                                    AbstractLiteral::Matrix(items, _) => {
267                                        let items = items
268                                            .into_iter()
269                                            .map(|x| match x {
270                                                Literal::Bool(false) => Literal::Int(0),
271                                                Literal::Bool(true) => Literal::Int(1),
272                                                x => x,
273                                            })
274                                            .collect_vec();
275
276                                        AbstractLiteral::Matrix(items, Domain::IntDomain(vec![]))
277                                    }
278                                    x => x,
279                                },
280                            ));
281                        updates.push((k, Literal::AbstractLiteral(matrix)));
282                    }
283                    _ => {}
284                }
285            }
286        }
287
288        for (k, v) in updates {
289            solset.insert(k, v);
290        }
291    }
292
293    // Remove duplicates
294    normalized = normalized.into_iter().unique().collect();
295    normalized
