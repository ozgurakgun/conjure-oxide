8pub fn expand_expr(essence: &TokenTree) -> Result<TokenStream> {
9    let src = to_src(essence);
10    let (tree, source_code) =
11        get_tree(&src).ok_or(Error::new(essence.span(), "Could not parse Essence AST"))?;
12    let root = tree.root_node();
13
14    // Get top level expressions
15    let mut query = query_toplevel(&root, &|n| n.kind() == "expression");
16    let expr_node = query
17        .next()
18        .ok_or(Error::new(essence.span(), "Expected an Essence expression"))?;
19
20    // We only expect one expression, error if that's not the case
21    if let Some(expr) = query.next() {
22        let tokens = &source_code[expr.start_byte()..expr.end_byte()];
23        return Err(Error::new(essence.span(), format!("Unexpected tokens: `{}`. Expected a single Essence expression. Perhaps you meant `essence_vec!`?", tokens)));
24    }
25
26    // Parse expression and build the token stream
27    let expr = mk_expr(expr_node, &source_code, &root, essence)?;
28    Ok(expr)
29}

31pub fn expand_expr_vec(tt: &TokenTree) -> Result<TokenStream> {
32    let mut ans: Vec<TokenStream> = Vec::new();
33    let src = to_src(tt);
34    let (tree, source_code) =
35        get_tree(&src).ok_or(Error::new(tt.span(), "Could not parse Essence AST"))?;
36    let root = tree.root_node();
37
38    let query = query_toplevel(&root, &|n| n.kind() == "expression");
39    for expr_node in query {
40        let expr = mk_expr(expr_node, &source_code, &root, tt)?;
41        ans.push(expr);
42    }
43    Ok(quote! { vec![#(#ans),*] })
44}

46/// Parse a single expression or make a compile time error
47fn mk_expr(node: Node, src: &str, root: &Node, tt: &TokenTree) -> Result<TokenStream> {
48    match parse_expr_to_ts(node, src, root) {
49        Ok(expr) => Ok(expr),
50        Err(err) => {
51            let msg = format!("Error parsing Essence expression: {}", err);
52            Err(Error::new(tt.span(), msg))
53        }
54    }
55}

57/// Parse string literals (gets rid of ""), otherwise use tokens as is
58fn to_src(tt: &TokenTree) -> String {
59    match syn::parse::<LitStr>(tt.to_token_stream().into()) {
60        Ok(src) => src.value(),
61        Err(_) => tt.to_string(),
62    }
