32pub fn model_from_json(str: &str, context: Arc<RwLock<Context<'static>>>) -> Result<Model> {
33    let mut m = Model::new(context);
34    let v: JsonValue = serde_json::from_str(str)?;
35    let statements = v["mStatements"]
36        .as_array()
37        .ok_or(error!("mStatements is not an array"))?;
38
39    for statement in statements {
40        let entry = statement
41            .as_object()
42            .ok_or(error!("mStatements contains a non-object"))?
43            .iter()
44            .next()
45            .ok_or(error!("mStatements contains an empty object"))?;
46
47        match entry.0.as_str() {
48            "Declaration" => {
49                let decl = entry
50                    .1
51                    .as_object()
52                    .ok_or(error!("Declaration is not an object".to_owned()))?;
53
54                // One field in the declaration should tell us what kind it is.
55                //
56                // Find it, ignoring the other fields.
57                //
58                // e.g. FindOrGiven,
59
60                let mut valid_decl: bool = false;
61                let scope = m.as_submodel().symbols_ptr_unchecked().clone();
62                let submodel = m.as_submodel_mut();
63                for (kind, value) in decl {
64                    match kind.as_str() {
65                        "FindOrGiven" => {
66                            parse_variable(value, &mut submodel.symbols_mut())?;
67                            valid_decl = true;
68                            break;
69                        }
70                        "Letting" => {
71                            parse_letting(value, &scope)?;
72                            valid_decl = true;
73                            break;
74                        }
75                        _ => continue,
76                    }
77                }
78
79                if !valid_decl {
80                    throw_error!("Declaration is not a valid kind")?;
81                }
82            }
83            "SuchThat" => {
84                let constraints_arr = match entry.1.as_array() {
85                    Some(x) => x,
86                    None => bug!("SuchThat is not a vector"),
87                };
88
89                let constraints: Vec<Expression> = constraints_arr
90                    .iter()
91                    .map(|x| {
92                        parse_expression(x, m.as_submodel_mut().symbols_ptr_unchecked()).unwrap()
93                    })
94                    .collect();
95                m.as_submodel_mut().add_constraints(constraints);
96                // println!("Nb constraints {}", m.constraints.len());
97            }
98            otherwise => bug!("Unhandled Statement {:#?}", otherwise),
99        }
100    }
101
102    Ok(m)
103}

105fn parse_variable(v: &JsonValue, symtab: &mut SymbolTable) -> Result<()> {
106    let arr = v.as_array().ok_or(error!("FindOrGiven is not an array"))?;
107    let name = arr[1]
108        .as_object()
109        .ok_or(error!("FindOrGiven[1] is not an object"))?["Name"]
110        .as_str()
111        .ok_or(error!("FindOrGiven[1].Name is not a string"))?;
112
113    let name = Name::UserName(name.to_owned());
114
115    let domain = arr[2]
116        .as_object()
117        .ok_or(error!("FindOrGiven[2] is not an object"))?
118        .iter()
119        .next()
120        .ok_or(error!("FindOrGiven[2] is an empty object"))?;
121
122    let domain = parse_domain(domain.0, domain.1, symtab)?;
123
124    symtab
125        .insert(Rc::new(Declaration::new_var(name.clone(), domain)))
126        .ok_or(Error::Parse(format!(
127            "Could not add {name} to symbol table as it already exists"
128        )))
129}

131fn parse_letting(v: &JsonValue, scope: &Rc<RefCell<SymbolTable>>) -> Result<()> {
132    let arr = v.as_array().ok_or(error!("Letting is not an array"))?;
133    let name = arr[0]
134        .as_object()
135        .ok_or(error!("Letting[0] is not an object"))?["Name"]
136        .as_str()
137        .ok_or(error!("Letting[0].Name is not a string"))?;
138    let name = Name::UserName(name.to_owned());
139    // value letting
140    if let Some(value) = parse_expression(&arr[1], scope) {
141        let mut symtab = scope.borrow_mut();
142        symtab
143            .insert(Rc::new(Declaration::new_value_letting(name.clone(), value)))
144            .ok_or(Error::Parse(format!(
145                "Could not add {name} to symbol table as it already exists"
146            )))
147    } else {
148        // domain letting
149        let domain = &arr[1]
150            .as_object()
151            .ok_or(error!("Letting[1] is not an object".to_owned()))?["Domain"]
152            .as_object()
153            .ok_or(error!("Letting[1].Domain is not an object"))?
154            .iter()
155            .next()
156            .ok_or(error!("Letting[1].Domain is an empty object"))?;
157
158        let mut symtab = scope.borrow_mut();
159        let domain = parse_domain(domain.0, domain.1, &symtab)?;
160
161        symtab
162            .insert(Rc::new(Declaration::new_domain_letting(
163                name.clone(),
164                domain,
165            )))
166            .ok_or(Error::Parse(format!(
167                "Could not add {name} to symbol table as it already exists"
168            )))
169    }
170}

172fn parse_domain(
173    domain_name: &str,
174    domain_value: &JsonValue,
175    symbols: &SymbolTable,
176) -> Result<Domain> {
177    match domain_name {
178        "DomainInt" => Ok(parse_int_domain(domain_value, symbols)?),
179        "DomainBool" => Ok(Domain::BoolDomain),
180        "DomainReference" => Ok(Domain::DomainReference(Name::UserName(
181            domain_value
182                .as_array()
183                .ok_or(error!("DomainReference is not an array"))?[0]
184                .as_object()
185                .ok_or(error!("DomainReference[0] is not an object"))?["Name"]
186                .as_str()
187                .ok_or(error!("DomainReference[0].Name is not a string"))?
188                .into(),
189        ))),
190        "DomainSet" => {
191            let dom = domain_value.get(2).and_then(|v| v.as_object());
192            let domain_obj = dom.expect("domain object exists");
193            let domain = domain_obj
194                .iter()
195                .next()
196                .ok_or(Error::Parse("DomainSet is an empty object".to_owned()))?;
197            let domain = match domain_name {
198                "DomainInt" => {
199                    println!("DomainInt: {:#?}", domain.1);
200                    Ok(parse_int_domain(domain.1, symbols)?)
201                }
202                "DomainBool" => Ok(Domain::BoolDomain),
203                _ => Err(Error::Parse(
204                    "FindOrGiven[2] is an unknown object".to_owned(),
205                )),
206            }?;
207            print!("{:?}", domain);
208            Ok(Domain::DomainSet(SetAttr::None, Box::new(domain)))
209        }
210
211        "DomainMatrix" => {
212            let domain_value = domain_value
213                .as_array()
214                .ok_or(error!("Domain matrix is not an array"))?;
215
216            let indexed_by_domain = domain_value[0].clone();
217            let (index_domain_name, index_domain_value) = indexed_by_domain
218                .as_object()
219                .ok_or(error!("DomainMatrix[0] is not an object"))?
220                .iter()
221                .next()
222                .ok_or(error!(""))?;
223
224            let (value_domain_name, value_domain_value) = domain_value[1]
225                .as_object()
226                .ok_or(error!(""))?
227                .iter()
228                .next()
229                .ok_or(error!(""))?;
230
231            // Conjure stores a 2-d matrix as a matrix of a matrix.
232            //
233            // Therefore, the index is always a Domain.
234
235            let mut index_domains: Vec<Domain> = vec![];
236
237            index_domains.push(parse_domain(
238                index_domain_name,
239                index_domain_value,
240                symbols,
241            )?);
242
243            // We want to store 2-d matrices as a matrix with two index domains, not a matrix in a
244            // matrix.
245            //
246            // Walk through the value domain until it is not a DomainMatrix, adding the index to
247            // our list of indices.
248            let mut value_domain = parse_domain(value_domain_name, value_domain_value, symbols)?;
249            while let Domain::DomainMatrix(new_value_domain, mut indices) = value_domain {
250                index_domains.append(&mut indices);
251                value_domain = *new_value_domain.clone()
252            }
253
254            Ok(Domain::DomainMatrix(Box::new(value_domain), index_domains))
255        }
256        "DomainTuple" => {
257            let domain_value = domain_value
258                .as_array()
259                .ok_or(error!("Domain matrix is not an array"))?;
260
261            //iterate through the array and parse each domain, should insert into the symbols table too
262            let domain = domain_value
263                .iter()
264                .map(|x| {
265                    let domain = x
266                        .as_object()
267                        .ok_or(error!("DomainTuple[0] is not an object"))?
268                        .iter()
269                        .next()
270                        .ok_or(error!("DomainTuple[0] is an empty object"))?;
271                    parse_domain(domain.0, domain.1, symbols)
272                })
273                .collect::<Result<Vec<Domain>>>()?;
274
275            Ok(Domain::DomainTuple(domain))
276        }
277
278        _ => Err(Error::Parse(
279            "FindOrGiven[2] is an unknown object".to_owned(), // consider covered
280        )),
281    }
282}

284fn parse_int_domain(v: &JsonValue, symbols: &SymbolTable) -> Result<Domain> {
285    let mut ranges = Vec::new();
286    let arr = v
287        .as_array()
288        .ok_or(error!("DomainInt is not an array".to_owned()))?[1]
289        .as_array()
290        .ok_or(error!("DomainInt[1] is not an array".to_owned()))?;
291    for range in arr {
292        let range = range
293            .as_object()
294            .ok_or(error!("DomainInt[1] contains a non-object"))?
295            .iter()
296            .next()
297            .ok_or(error!("DomainInt[1] contains an empty object"))?;
298        match range.0.as_str() {
299            "RangeBounded" => {
300                let arr = range
301                    .1
302                    .as_array()
303                    .ok_or(error!("RangeBounded is not an array".to_owned()))?;
304                let mut nums = Vec::new();
305                for item in arr.iter() {
306                    let num = parse_domain_value_int(item, symbols)
307                        .ok_or(error!("Could not parse int domain constant"))?;
308                    nums.push(num);
309                }
310                ranges.push(Range::Bounded(nums[0], nums[1]));
311            }
312            "RangeSingle" => {
313                let num = parse_domain_value_int(range.1, symbols)
314                    .ok_or(error!("Could not parse int domain constant"))?;
315                ranges.push(Range::Single(num));
316            }
317            _ => return throw_error!("DomainInt[1] contains an unknown object"),
318        }
319    }
320    Ok(Domain::IntDomain(ranges))
321}

323/// Parses a (possibly) integer value inside the range of a domain
324///
325/// 1. (positive number) Constant/ConstantInt/1
326///
327/// 2. (negative number) Op/MkOpNegate/Constant/ConstantInt/1
328///
329/// Unlike `parse_constant` this handles the negation operator. `parse_constant` expects the
330/// negation to already have been handled as an expression; however, here we do not expect domain
331/// values to be part of larger expressions, only negated.
332///
333fn parse_domain_value_int(obj: &JsonValue, symbols: &SymbolTable) -> Option<i32> {
334    parser_trace!("trying to parse domain value: {}", obj);
335
336    fn try_parse_positive_int(obj: &JsonValue) -> Option<i32> {
337        parser_trace!(".. trying as a positive domain value: {}", obj);
338        // Positive number: Constant/ConstantInt/1
339
340        let leaf_node = obj
341            .pointer("/Constant/ConstantInt/1")
342            .or_else(|| obj.pointer("/ConstantInt/1"))?;
343
344        match leaf_node.as_i64()?.try_into() {
345            Ok(x) => {
346                parser_trace!(".. success!");
347                Some(x)
348            }
349            Err(_) => {
350                println!(
351                    "Could not convert integer constant to i32: {:#?}",
352                    leaf_node
353                );
354                None
355            }
356        }
357    }
358
359    fn try_parse_negative_int(obj: &JsonValue) -> Option<i32> {
360        // Negative number: Op/MkOpNegate/Constant/ConstantInt/1
361
362        // Unwrap negation operator, giving us a Constant/ConstantInt/1
363        //
364        // This is just a positive constant, so try to parse it as such
365
366        parser_trace!(".. trying as a negative domain value: {}", obj);
367        let inner_constant_node = obj.pointer("/Op/MkOpNegate")?;
368        let inner_num = try_parse_positive_int(inner_constant_node)?;
369
370        parser_trace!(".. success!");
371        Some(-inner_num)
372    }
373
374    // it will be too annoying to store references inside all our int domain ranges, so just
375    // resolve them here.
376    //
377    // this matches savilerow, where lettings must be declared before they are used.
378    //
379    // TODO: we shouldn't do this long term, add support for ranges containing domain references.
380    fn try_parse_reference(obj: &JsonValue, symbols: &SymbolTable) -> Option<i32> {
381        parser_trace!(".. trying as a domain reference: {}", obj);
382        let inner_name = obj.pointer("/Reference/0/Name")?.as_str()?;
383        parser_trace!(
384            ".. found domain reference to {}, trying to resolve it",
385            inner_name
386        );
387        let name = Name::UserName(inner_name.to_string());
388        let decl = symbols.lookup(&name)?;
389        let DeclarationKind::ValueLetting(d) = decl.kind() else {
390            parser_trace!(".. name exists but is not a value letting!");
391            return None;
392        };
393
394        let a = d.clone().to_literal()?;
395        let Literal::Int(a) = a else {
396            return None;
397        };
398
399        Some(a)
400    }
401
402    try_parse_positive_int(obj)
403        .or_else(|| try_parse_negative_int(obj))
404        .or_else(|| try_parse_reference(obj, symbols))
405}

407// this needs an explicit type signature to force the closures to have the same type
408type BinOp = Box<dyn Fn(Metadata, Box<Expression>, Box<Expression>) -> Expression>;
409type UnaryOp = Box<dyn Fn(Metadata, Box<Expression>) -> Expression>;
410type VecOp = Box<dyn Fn(Metadata, Vec<Expression>) -> Expression>;

412pub fn parse_expression(obj: &JsonValue, scope: &Rc<RefCell<SymbolTable>>) -> Option<Expression> {
413    let binary_operators: HashMap<&str, BinOp> = [
414        (
415            "MkOpEq",
416            Box::new(Expression::Eq) as Box<dyn Fn(_, _, _) -> _>,
417        ),
418        (
419            "MkOpNeq",
420            Box::new(Expression::Neq) as Box<dyn Fn(_, _, _) -> _>,
421        ),
422        (
423            "MkOpGeq",
424            Box::new(Expression::Geq) as Box<dyn Fn(_, _, _) -> _>,
425        ),
426        (
427            "MkOpLeq",
428            Box::new(Expression::Leq) as Box<dyn Fn(_, _, _) -> _>,
429        ),
430        (
431            "MkOpGt",
432            Box::new(Expression::Gt) as Box<dyn Fn(_, _, _) -> _>,
433        ),
434        (
435            "MkOpLt",
436            Box::new(Expression::Lt) as Box<dyn Fn(_, _, _) -> _>,
437        ),
438        (
439            "MkOpGt",
440            Box::new(Expression::Gt) as Box<dyn Fn(_, _, _) -> _>,
441        ),
442        (
443            "MkOpLt",
444            Box::new(Expression::Lt) as Box<dyn Fn(_, _, _) -> _>,
445        ),
446        (
447            "MkOpDiv",
448            Box::new(Expression::UnsafeDiv) as Box<dyn Fn(_, _, _) -> _>,
449        ),
450        (
451            "MkOpMod",
452            Box::new(Expression::UnsafeMod) as Box<dyn Fn(_, _, _) -> _>,
453        ),
454        (
455            "MkOpMinus",
456            Box::new(Expression::Minus) as Box<dyn Fn(_, _, _) -> _>,
457        ),
458        (
459            "MkOpImply",
460            Box::new(Expression::Imply) as Box<dyn Fn(_, _, _) -> _>,
461        ),
462        (
463            "MkOpPow",
464            Box::new(Expression::UnsafePow) as Box<dyn Fn(_, _, _) -> _>,
465        ),
466    ]
467    .into_iter()
468    .collect();
469
470    let unary_operators: HashMap<&str, UnaryOp> = [
471        (
472            "MkOpNot",
473            Box::new(Expression::Not) as Box<dyn Fn(_, _) -> _>,
474        ),
475        (
476            "MkOpNegate",
477            Box::new(Expression::Neg) as Box<dyn Fn(_, _) -> _>,
478        ),
479        (
480            "MkOpTwoBars",
481            Box::new(Expression::Abs) as Box<dyn Fn(_, _) -> _>,
482        ),
483        (
484            "MkOpAnd",
485            Box::new(Expression::And) as Box<dyn Fn(_, _) -> _>,
486        ),
487        (
488            "MkOpSum",
489            Box::new(Expression::Sum) as Box<dyn Fn(_, _) -> _>,
490        ),
491        ("MkOpOr", Box::new(Expression::Or) as Box<dyn Fn(_, _) -> _>),
492        (
493            "MkOpMin",
494            Box::new(Expression::Min) as Box<dyn Fn(_, _) -> _>,
495        ),
496        (
497            "MkOpMax",
498            Box::new(Expression::Max) as Box<dyn Fn(_, _) -> _>,
499        ),
500        (
501            "MkOpAllDiff",
502            Box::new(Expression::AllDiff) as Box<dyn Fn(_, _) -> _>,
503        ),
504    ]
505    .into_iter()
506    .collect();
507
508    let vec_operators: HashMap<&str, VecOp> = [(
509        "MkOpProduct",
510        Box::new(Expression::Product) as Box<dyn Fn(_, _) -> _>,
511    )]
512    .into_iter()
513    .collect();
514
515    let mut binary_operator_names = binary_operators.iter().map(|x| x.0);
516    let mut unary_operator_names = unary_operators.iter().map(|x| x.0);
517    let mut vec_operator_names = vec_operators.iter().map(|x| x.0);
518    #[allow(clippy::unwrap_used)]
519    match obj {
520        Value::Object(op) if op.contains_key("Op") => match &op["Op"] {
521            Value::Object(bin_op) if binary_operator_names.any(|key| bin_op.contains_key(*key)) => {
522                Some(parse_bin_op(bin_op, binary_operators, scope).unwrap())
523            }
524            Value::Object(un_op) if unary_operator_names.any(|key| un_op.contains_key(*key)) => {
525                Some(parse_unary_op(un_op, unary_operators, scope).unwrap())
526            }
527            Value::Object(vec_op) if vec_operator_names.any(|key| vec_op.contains_key(*key)) => {
528                Some(parse_vec_op(vec_op, vec_operators, scope).unwrap())
529            }
530
531            Value::Object(op)
532                if op.contains_key("MkOpIndexing") || op.contains_key("MkOpSlicing") =>
533            {
534                parse_indexing_slicing_op(op, scope)
535            }
536            otherwise => bug!("Unhandled Op {:#?}", otherwise),
537        },
538        Value::Object(comprehension) if comprehension.contains_key("Comprehension") => {
539            Some(parse_comprehension(comprehension, Rc::clone(scope), None).unwrap())
540        }
541        Value::Object(refe) if refe.contains_key("Reference") => {
542            let name = refe["Reference"].as_array()?[0].as_object()?["Name"].as_str()?;
543            Some(Expression::Atomic(
544                Metadata::new(),
545                Atom::Reference(Name::UserName(name.to_string())),
546            ))
547        }
548        Value::Object(abslit) if abslit.contains_key("AbstractLiteral") => {
549            if abslit["AbstractLiteral"]
550                .as_object()?
551                .contains_key("AbsLitSet")
552            {
553                Some(parse_abs_lit(&abslit["AbstractLiteral"]["AbsLitSet"], scope).unwrap())
554            } else {
555                Some(parse_abstract_matrix_as_expr(obj, scope).unwrap())
556            }
557        }
558
559        Value::Object(constant) if constant.contains_key("Constant") => Some(
560            parse_constant(constant, scope)
561                .or_else(|| parse_abstract_matrix_as_expr(obj, scope))
562                .unwrap(),
563        ),
564
565        Value::Object(constant) if constant.contains_key("ConstantAbstract") => {
566            Some(parse_abstract_matrix_as_expr(obj, scope).unwrap())
567        }
568
569        Value::Object(constant) if constant.contains_key("ConstantInt") => {
570            Some(parse_constant(constant, scope).unwrap())
571        }
572        Value::Object(constant) if constant.contains_key("ConstantBool") => {
573            Some(parse_constant(constant, scope).unwrap())
574        }
575
576        _ => None,
577    }
578}

580fn parse_abs_lit(abs_set: &Value, scope: &Rc<RefCell<SymbolTable>>) -> Option<Expression> {
581    let values = abs_set.as_array()?; // Ensure it's an array
582    let expressions = values
583        .iter()
584        .map(|values| parse_expression(values, scope))
585        .map(|values| values.expect("invalid subexpression")) // Ensure valid expressions
586        .collect::<Vec<Expression>>(); // Collect all expressions
587
588    Some(Expression::AbstractLiteral(
589        Metadata::new(),
590        AbstractLiteral::Set(expressions),
591    ))
592}

608fn parse_comprehension(
609    comprehension: &serde_json::Map<String, Value>,
610    scope: Rc<RefCell<SymbolTable>>,
611    comprehension_kind: Option<ComprehensionKind>,
612) -> Option<Expression> {
613    let value = &comprehension["Comprehension"];
614    let mut comprehension = ComprehensionBuilder::new();
615    let expr = parse_expression(value.pointer("/0")?, &scope)?;
616
617    let generators_and_guards = value.pointer("/1")?.as_array()?.iter();
618
619    for value in generators_and_guards {
620        let value = value.as_object()?;
621        let (name, value) = value.iter().next()?;
622        comprehension = match name.as_str() {
623            "Generator" => {
624                // TODO: more things than GenDomainNoRepr and Single names here?
625                let name = value.pointer("/GenDomainNoRepr/0/Single/Name")?.as_str()?;
626                let (domain_name, domain_value) = value
627                    .pointer("/GenDomainNoRepr/1")?
628                    .as_object()?
629                    .iter()
630                    .next()?;
631                let domain = parse_domain(domain_name, domain_value, &scope.borrow()).ok()?;
632                comprehension.generator(Name::UserName(name.to_string()), domain)
633            }
634
635            "Condition" => comprehension.guard(parse_expression(value, &scope)?),
636
637            x => {
638                bug!("unknown field inside comprehension {x}");
639            }
640        }
641    }
642
643    Some(Expression::Comprehension(
644        Metadata::new(),
645        Box::new(comprehension.with_return_value(expr, scope, comprehension_kind)),
646    ))
647}

665        }
666        otherwise => bug!("Unhandled parse_bin_op {:#?}", otherwise),
667    }

670fn parse_indexing_slicing_op(
671    op: &serde_json::Map<String, Value>,
672    scope: &Rc<RefCell<SymbolTable>>,
673) -> Option<Expression> {
674    // we know there is a single key value pair in this object
675    // extract the value, ignore the key
676    let (key, value) = op.into_iter().next()?;
677
678    // we know that this is meant to be a mkopindexing, so anything that goes wrong from here is a
679    // bug!
680
681    // Conjure does a[1,2,3] as MkOpIndexing(MkOpIndexing(MkOpIndexing(a,3),2),1).
682    //
683    // And  a[1,..,3] as MkOpIndexing(MkOpSlicing(MkOpIndexing(a,3)),1).
684    //
685    // However, we want this in a flattened form: Index(a, [1,2,3])
686    let mut target: Expression;
687    let mut indices: Vec<Option<Expression>> = vec![];
688
689    // true if this has no slicing, false otherwise.
690    let mut all_known = true;
691
692    match key.as_str() {
693        "MkOpIndexing" => {
694            match &value {
695                Value::Array(op_args) if op_args.len() == 2 => {
696                    target = parse_expression(&op_args[0], scope).expect("expected an expression");
697                    indices.push(Some(
698                        parse_expression(&op_args[1], scope).expect("expected an expression"),
699                    ));
700                }
701                otherwise => bug!("Unknown object inside MkOpIndexing: {:#?}", otherwise),
702            };
703        }
704
705        "MkOpSlicing" => {
706            all_known = false;
707            match &value {
708                Value::Array(op_args) if op_args.len() == 3 => {
709                    target = parse_expression(&op_args[0], scope).expect("expected an expression");
710                    indices.push(None);
711                }
712                otherwise => bug!("Unknown object inside MkOpSlicing: {:#?}", otherwise),
713            };
714        }
715
716        _ => {
717            return None;
718        }
719    }
720
721    loop {
722        match &mut target {
723            Expression::UnsafeIndex(_, new_target, new_indices) => {
724                indices.extend(new_indices.iter().cloned().map(Some));
725                target = *new_target.clone();
726            }
727
728            Expression::UnsafeSlice(_, new_target, new_indices) => {
729                all_known = false;
730                indices.append(new_indices);
731                target = *new_target.clone();
732            }
733
734            _ => {
735                // not a slice or an index, we have reached the target.
736                break;
737            }
738        }
739    }
740
741    indices.reverse();
742
743    if all_known {
744        Some(Expression::UnsafeIndex(
745            Metadata::new(),
746            Box::new(target),
747            indices.into_iter().map(|x| x.unwrap()).collect(),
748        ))
749    } else {
750        Some(Expression::UnsafeSlice(
751            Metadata::new(),
752            Box::new(target),
753            indices,
754        ))
755    }
756}

758fn parse_unary_op(
759    un_op: &serde_json::Map<String, Value>,
760    unary_operators: HashMap<&str, UnaryOp>,
761    scope: &Rc<RefCell<SymbolTable>>,
762) -> Option<Expression> {
763    let (key, value) = un_op.into_iter().next()?;
764    let constructor = unary_operators.get(key.as_str())?;
765
766    // unops are the main things that contain comprehensions
767    //
768    // if the current expr is a quantifier like and/or/sum and it contains a comprehension, let the comprehension know what it is inside.
769    let arg = match value {
770        Value::Object(comprehension) if comprehension.contains_key("Comprehension") => {
771            let comprehension_kind = match key.as_str() {
772                "MkOpOr" => Some(ComprehensionKind::Or),
773                "MkOpAnd" => Some(ComprehensionKind::And),
774                "MkOpSum" => Some(ComprehensionKind::Sum),
775                _ => None,
776            };
777            Some(parse_comprehension(comprehension, Rc::clone(scope), comprehension_kind).unwrap())
778        }
779        _ => parse_expression(value, scope),
780    }?;
781
782    Some(constructor(Metadata::new(), Box::new(arg)))
783}

785fn parse_vec_op(
786    vec_op: &serde_json::Map<String, Value>,
787    vec_operators: HashMap<&str, VecOp>,
788    scope: &Rc<RefCell<SymbolTable>>,
789) -> Option<Expression> {
790    let (key, value) = vec_op.into_iter().next()?;
791    let constructor = vec_operators.get(key.as_str())?;
792
793    parser_debug!("Trying to parse vec_op: {key} ...");
794
795    let mut args_parsed: Option<Vec<Option<Expression>>> = None;
796    if let Some(abs_lit_matrix) = value.pointer("/AbstractLiteral/AbsLitMatrix/1") {
797        parser_trace!("... containing a matrix of literals");
798        args_parsed = abs_lit_matrix.as_array().map(|x| {
799            x.iter()
800                .map(|x| parse_expression(x, scope))
801                .collect::<Vec<Option<Expression>>>()
802        });
803    }
804    // the input of this expression is constant - e.g. or([]), or([false]), min([2]), etc.
805    else if let Some(const_abs_lit_matrix) =
806        value.pointer("/Constant/ConstantAbstract/AbsLitMatrix/1")
807    {
808        parser_trace!("... containing a matrix of constants");
809        args_parsed = const_abs_lit_matrix.as_array().map(|x| {
810            x.iter()
811                .map(|x| parse_expression(x, scope))
812                .collect::<Vec<Option<Expression>>>()
813        });
814    }
815
816    let args_parsed = args_parsed?;
817
818    let number_of_args = args_parsed.len();
819    parser_debug!("... with {number_of_args} args {args_parsed:#?}");
820
821    let valid_args: Vec<Expression> = args_parsed.into_iter().flatten().collect();
822    if number_of_args != valid_args.len() {
823        None
824    } else {
825        parser_debug!("... success!");
826        Some(constructor(Metadata::new(), valid_args))
827    }
828}

830// Takes in { AbstractLiteral: .... }
831fn parse_abstract_matrix_as_expr(
832    value: &serde_json::Value,
833    scope: &Rc<RefCell<SymbolTable>>,
834) -> Option<Expression> {
835    parser_trace!("trying to parse an abstract literal matrix");
836    let (values, domain_name, domain_value) =
837        if let Some(abs_lit_matrix) = value.pointer("/AbstractLiteral/AbsLitMatrix") {
838            parser_trace!(".. found JSON pointer /AbstractLiteral/AbstractLitMatrix");
839            let (domain_name, domain_value) =
840                abs_lit_matrix.pointer("/0")?.as_object()?.iter().next()?;
841            let values = abs_lit_matrix.pointer("/1")?;
842
843            Some((values, domain_name, domain_value))
844        }
845        // the input of this expression is constant - e.g. or([]), or([false]), min([2]), etc.
846        else if let Some(const_abs_lit_matrix) =
847            value.pointer("/Constant/ConstantAbstract/AbsLitMatrix")
848        {
849            parser_trace!(".. found JSON pointer /Constant/ConstantAbstract/AbsLitMatrix");
850            let (domain_name, domain_value) = const_abs_lit_matrix
851                .pointer("/0")?
852                .as_object()?
853                .iter()
854                .next()?;
855            let values = const_abs_lit_matrix.pointer("/1")?;
856
857            Some((values, domain_name, domain_value))
858        } else if let Some(const_abs_lit_matrix) = value.pointer("/ConstantAbstract/AbsLitMatrix") {
859            parser_trace!(".. found JSON pointer /ConstantAbstract/AbsLitMatrix");
860            let (domain_name, domain_value) = const_abs_lit_matrix
861                .pointer("/0")?
862                .as_object()?
863                .iter()
864                .next()?;
865            let values = const_abs_lit_matrix.pointer("/1")?;
866            Some((values, domain_name, domain_value))
867        } else {
868            None
869        }?;
870
871    parser_trace!(".. found in domain and values in JSON:");
872    parser_trace!(".. .. index domain name {domain_name}");
873    parser_trace!(".. .. values {value}");
874
875    let args_parsed = values.as_array().map(|x| {
876        x.iter()
877            .map(|x| parse_expression(x, scope))
878            .map(|x| x.expect("invalid subexpression"))
879            .collect::<Vec<Expression>>()
880    })?;
881
882    if !args_parsed.is_empty() {
883        parser_trace!(
884            ".. successfully parsed values as expressions: {}, ... ",
885            args_parsed[0]
886        );
887    } else {
888        parser_trace!(".. successfully parsed empty values ",);
889    }
890
891    let symbols = scope.borrow();
892    match parse_domain(domain_name, domain_value, &symbols) {
893        Ok(domain) => {
894            parser_trace!("... sucessfully parsed domain as {domain}");
895            Some(into_matrix_expr![args_parsed;domain])
896        }
897        Err(_) => {
898            parser_trace!("... failed to parse domain, creating a matrix without one.");
899            Some(into_matrix_expr![args_parsed])
900        }
901    }
902}

904fn parse_constant(
905    constant: &serde_json::Map<String, Value>,
906    scope: &Rc<RefCell<SymbolTable>>,
907) -> Option<Expression> {
908    match &constant.get("Constant") {
909        Some(Value::Object(int)) if int.contains_key("ConstantInt") => {
910            let int_32: i32 = match int["ConstantInt"].as_array()?[1].as_i64()?.try_into() {
911                Ok(x) => x,
912                Err(_) => {
913                    println!(
914                        "Could not convert integer constant to i32: {:#?}",
915                        int["ConstantInt"]
916                    );
917                    return None;
918                }
919            };
920
921            Some(Expression::Atomic(
922                Metadata::new(),
923                Atom::Literal(Literal::Int(int_32)),
924            ))
925        }
926
927        Some(Value::Object(b)) if b.contains_key("ConstantBool") => {
928            let b: bool = b["ConstantBool"].as_bool()?;
929            Some(Expression::Atomic(
930                Metadata::new(),
931                Atom::Literal(Literal::Bool(b)),
932            ))
933        }
934
935        Some(Value::Object(int)) if int.contains_key("ConstantAbstract") => {
936            if let Some(Value::Object(obj)) = int.get("ConstantAbstract") {
937                if let Some(arr) = obj.get("AbsLitSet") {
938                    return parse_abs_lit(arr, scope);
939                } else if let Some(arr) = obj.get("AbsLitMatrix") {
940                    return parse_abstract_matrix_as_expr(arr, scope);
941                } else if let Some(arr) = obj.get("AbsLitTuple") {
942                    return parse_abs_tuple(arr, scope);
943                }
944            }
945            None
946        }
947
948        // sometimes (e.g. constant matrices) we can have a ConstantInt / Constant bool that is
949        // not wrapped in Constant
950        None => {
951            let int_expr = constant
952                .get("ConstantInt")
953                .and_then(|x| x.as_array())
954                .and_then(|x| x[1].as_i64())
955                .and_then(|x| x.try_into().ok())
956                .map(|x| Expression::Atomic(Metadata::new(), Atom::Literal(Literal::Int(x))));
957
958            if let e @ Some(_) = int_expr {
959                return e;
960            }
961
962            let bool_expr = constant
963                .get("ConstantBool")
964                .and_then(|x| x.as_bool())
965                .map(|x| Expression::Atomic(Metadata::new(), Atom::Literal(Literal::Bool(x))));
966
967            if let e @ Some(_) = bool_expr {
968                return e;
969            }
970
971            bug!("Unhandled parse_constant {:#?}", constant);
972        }
973        otherwise => bug!("Unhandled parse_constant {:#?}", otherwise),
974    }
975}
