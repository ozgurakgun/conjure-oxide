387fn expr_vec_to_domain_i32(
388    exprs: &[Expression],
389    op: fn(i32, i32) -> Option<i32>,
390    vars: &SymbolTable,
391) -> Option<Domain> {
392    let domains: Vec<Option<_>> = exprs.iter().map(|e| e.domain_of(vars)).collect();
393    domains
394        .into_iter()
395        .reduce(|a, b| a.and_then(|x| b.and_then(|y| x.apply_i32(op, &y))))
396        .flatten()
397}
398fn expr_vec_lit_to_domain_i32(
399    e: &Expression,
400    op: fn(i32, i32) -> Option<i32>,
401    vars: &SymbolTable,
402) -> Option<Domain> {
403    let exprs = e.clone().unwrap_list()?;
404    expr_vec_to_domain_i32(&exprs, op, vars)
405}

407// Returns none if unbounded
408fn range_vec_bounds_i32(ranges: &Vec<Range<i32>>) -> Option<(i32, i32)> {
409    let mut min = i32::MAX;
410    let mut max = i32::MIN;
411    for r in ranges {
412        match r {
413            Range::Single(i) => {
414                if *i < min {
415                    min = *i;
416                }
417                if *i > max {
418                    max = *i;
419                }
420            }
421            Range::Bounded(i, j) => {
422                if *i < min {
423                    min = *i;
424                }
425                if *j > max {
426                    max = *j;
427                }
428            }
429            Range::UnboundedR(_) | Range::UnboundedL(_) => return None,
430        }
431    }
432    Some((min, max))
433}

435impl Expression {
436    /// Returns the possible values of the expression, recursing to leaf expressions
437    pub fn domain_of(&self, syms: &SymbolTable) -> Option<Domain> {
438        let ret = match self {
439            //todo
440            Expression::AbstractLiteral(_, _) => None,
441            Expression::DominanceRelation(_, _) => Some(Domain::BoolDomain),
442            Expression::FromSolution(_, expr) => expr.domain_of(syms),
443            Expression::Comprehension(_, comprehension) => comprehension.domain_of(),
444            Expression::UnsafeIndex(_, matrix, _) | Expression::SafeIndex(_, matrix, _) => {
445                match matrix.domain_of(syms)? {
446                    Domain::DomainMatrix(elem_domain, _) => Some(*elem_domain),
447                    Domain::DomainTuple(_) => None,
448                    _ => {
449                        bug!("subject of an index operation should support indexing")
450                    }
451                }
452            }
453            Expression::UnsafeSlice(_, matrix, indices)
454            | Expression::SafeSlice(_, matrix, indices) => {
455                let sliced_dimension = indices.iter().position(Option::is_none);
456
457                let Domain::DomainMatrix(elem_domain, index_domains) = matrix.domain_of(syms)?
458                else {
459                    bug!("subject of an index operation should be a matrix");
460                };
461
462                match sliced_dimension {
463                    Some(dimension) => Some(Domain::DomainMatrix(
464                        elem_domain,
465                        vec![index_domains[dimension].clone()],
466                    )),
467
468                    // same as index
469                    None => Some(*elem_domain),
470                }
471            }
472            Expression::InDomain(_, _, _) => Some(Domain::BoolDomain),
473            Expression::Atomic(_, Atom::Reference(name)) => Some(syms.resolve_domain(name)?),
474            Expression::Atomic(_, Atom::Literal(Literal::Int(n))) => {
475                Some(Domain::IntDomain(vec![Range::Single(*n)]))
476            }
477            Expression::Atomic(_, Atom::Literal(Literal::Bool(_))) => Some(Domain::BoolDomain),
478            Expression::Atomic(_, Atom::Literal(Literal::AbstractLiteral(_))) => None,
479            Expression::Scope(_, _) => Some(Domain::BoolDomain),
480            Expression::Sum(_, e) => expr_vec_lit_to_domain_i32(e, |x, y| Some(x + y), syms),
481            Expression::Product(_, exprs) => {
482                expr_vec_to_domain_i32(exprs, |x, y| Some(x * y), syms)
483            }
484            Expression::Min(_, e) => {
485                expr_vec_lit_to_domain_i32(e, |x, y| Some(if x < y { x } else { y }), syms)
486            }
487            Expression::Max(_, e) => {
488                expr_vec_lit_to_domain_i32(e, |x, y| Some(if x > y { x } else { y }), syms)
489            }
490            Expression::UnsafeDiv(_, a, b) => a.domain_of(syms)?.apply_i32(
491                // rust integer division is truncating; however, we want to always round down,
492                // including for negative numbers.
493                |x, y| {
494                    if y != 0 {
495                        Some((x as f32 / y as f32).floor() as i32)
496                    } else {
497                        None
498                    }
499                },
500                &b.domain_of(syms)?,
501            ),
502            Expression::SafeDiv(_, a, b) => {
503                // rust integer division is truncating; however, we want to always round down
504                // including for negative numbers.
505                let domain = a.domain_of(syms)?.apply_i32(
506                    |x, y| {
507                        if y != 0 {
508                            Some((x as f32 / y as f32).floor() as i32)
509                        } else {
510                            None
511                        }
512                    },
513                    &b.domain_of(syms)?,
514                );
515
516                match domain {
517                    Some(Domain::IntDomain(ranges)) => {
518                        let mut ranges = ranges;
519                        ranges.push(Range::Single(0));
520                        Some(Domain::IntDomain(ranges))
521                    }
522                    None => Some(Domain::IntDomain(vec![Range::Single(0)])),
523                    _ => None,
524                }
525            }
526            Expression::UnsafeMod(_, a, b) => a.domain_of(syms)?.apply_i32(
527                |x, y| if y != 0 { Some(x % y) } else { None },
528                &b.domain_of(syms)?,
529            ),
530
531            Expression::SafeMod(_, a, b) => {
532                let domain = a.domain_of(syms)?.apply_i32(
533                    |x, y| if y != 0 { Some(x % y) } else { None },
534                    &b.domain_of(syms)?,
535                );
536
537                match domain {
538                    Some(Domain::IntDomain(ranges)) => {
539                        let mut ranges = ranges;
540                        ranges.push(Range::Single(0));
541                        Some(Domain::IntDomain(ranges))
542                    }
543                    None => Some(Domain::IntDomain(vec![Range::Single(0)])),
544                    _ => None,
545                }
546            }
547
548            Expression::SafePow(_, a, b) | Expression::UnsafePow(_, a, b) => {
549                a.domain_of(syms)?.apply_i32(
550                    |x, y| {
551                        if (x != 0 || y != 0) && y >= 0 {
552                            Some(x.pow(y as u32))
553                        } else {
554                            None
555                        }
556                    },
557                    &b.domain_of(syms)?,
558                )
559            }
560
561            Expression::Root(_, _) => None,
562            Expression::Bubble(_, _, _) => None,
563            Expression::AuxDeclaration(_, _, _) => Some(Domain::BoolDomain),
564            Expression::And(_, _) => Some(Domain::BoolDomain),
565            Expression::Not(_, _) => Some(Domain::BoolDomain),
566            Expression::Or(_, _) => Some(Domain::BoolDomain),
567            Expression::Imply(_, _, _) => Some(Domain::BoolDomain),
568            Expression::Eq(_, _, _) => Some(Domain::BoolDomain),
569            Expression::Neq(_, _, _) => Some(Domain::BoolDomain),
570            Expression::Geq(_, _, _) => Some(Domain::BoolDomain),
571            Expression::Leq(_, _, _) => Some(Domain::BoolDomain),
572            Expression::Gt(_, _, _) => Some(Domain::BoolDomain),
573            Expression::Lt(_, _, _) => Some(Domain::BoolDomain),
574            Expression::FlatAbsEq(_, _, _) => Some(Domain::BoolDomain),
575            Expression::FlatSumGeq(_, _, _) => Some(Domain::BoolDomain),
576            Expression::FlatSumLeq(_, _, _) => Some(Domain::BoolDomain),
577            Expression::MinionDivEqUndefZero(_, _, _, _) => Some(Domain::BoolDomain),
578            Expression::MinionModuloEqUndefZero(_, _, _, _) => Some(Domain::BoolDomain),
579            Expression::FlatIneq(_, _, _, _) => Some(Domain::BoolDomain),
580            Expression::AllDiff(_, _) => Some(Domain::BoolDomain),
581            Expression::FlatWatchedLiteral(_, _, _) => Some(Domain::BoolDomain),
582            Expression::MinionReify(_, _, _) => Some(Domain::BoolDomain),
583            Expression::MinionReifyImply(_, _, _) => Some(Domain::BoolDomain),
584            Expression::MinionWInIntervalSet(_, _, _) => Some(Domain::BoolDomain),
585            Expression::MinionElementOne(_, _, _, _) => Some(Domain::BoolDomain),
586            Expression::Neg(_, x) => {
587                let Some(Domain::IntDomain(mut ranges)) = x.domain_of(syms) else {
588                    return None;
589                };
590
591                for range in ranges.iter_mut() {
592                    *range = match range {
593                        Range::Single(x) => Range::Single(-*x),
594                        Range::Bounded(x, y) => Range::Bounded(-*y, -*x),
595                        Range::UnboundedR(i) => Range::UnboundedL(-*i),
596                        Range::UnboundedL(i) => Range::UnboundedR(-*i),
597                    };
598                }
599
600                Some(Domain::IntDomain(ranges))
601            }
602            Expression::Minus(_, a, b) => a
603                .domain_of(syms)?
604                .apply_i32(|x, y| Some(x - y), &b.domain_of(syms)?),
605
606            Expression::FlatAllDiff(_, _) => Some(Domain::BoolDomain),
607            Expression::FlatMinusEq(_, _, _) => Some(Domain::BoolDomain),
608            Expression::FlatProductEq(_, _, _, _) => Some(Domain::BoolDomain),
609            Expression::FlatWeightedSumLeq(_, _, _, _) => Some(Domain::BoolDomain),
610            Expression::FlatWeightedSumGeq(_, _, _, _) => Some(Domain::BoolDomain),
611            Expression::Abs(_, a) => a
612                .domain_of(syms)?
613                .apply_i32(|a, _| Some(a.abs()), &a.domain_of(syms)?),
614            Expression::MinionPow(_, _, _, _) => Some(Domain::BoolDomain),
615        };
616        match ret {
617            // TODO: (flm8) the Minion bindings currently only support single ranges for domains, so we use the min/max bounds
618            // Once they support a full domain as we define it, we can remove this conversion
619            Some(Domain::IntDomain(ranges)) if ranges.len() > 1 => {
620                let (min, max) = range_vec_bounds_i32(&ranges)?;
621                Some(Domain::IntDomain(vec![Range::Bounded(min, max)]))
622            }
623            _ => ret,
624        }
625    }

636    /// Checks whether this expression is safe.
637    ///
638    /// An expression is unsafe if can be undefined, or if any of its children can be undefined.
639    ///
640    /// Unsafe expressions are (typically) prefixed with Unsafe in our AST, and can be made
641    /// safe through the use of bubble rules.
642    pub fn is_safe(&self) -> bool {
643        // TODO: memoise in Metadata
644        for expr in self.universe() {
645            match expr {
646                Expression::UnsafeDiv(_, _, _)
647                | Expression::UnsafeMod(_, _, _)
648                | Expression::UnsafePow(_, _, _)
649                | Expression::UnsafeIndex(_, _, _)
650                | Expression::UnsafeSlice(_, _, _) => {
651                    return false;
652                }
653                _ => {}
654            }
655        }
656        true
657    }

746    /// True iff self and other are both atomic and identical.
747    ///
748    /// This method is useful to cheaply check equivalence. Assuming CSE is enabled, any unifiable
749    /// expressions will be rewritten to a common variable. This is much cheaper than checking the
750    /// entire subtrees of `self` and `other`.
751    pub fn identical_atom_to(&self, other: &Expression) -> bool {
752        let atom1: Result<&Atom, _> = self.try_into();
753        let atom2: Result<&Atom, _> = other.try_into();
754
755        if let (Ok(atom1), Ok(atom2)) = (atom1, atom2) {
756            atom2 == atom1
757        } else {
758            false
759        }
760    }

762    /// If the expression is a list, returns the inner expressions.
763    ///
764    /// A list is any a matrix with the domain `int(1..)`. This includes matrix literals without
765    /// any explicitly specified domain.
766    pub fn unwrap_list(self) -> Option<Vec<Expression>> {
767        match self {
768            Expression::AbstractLiteral(_, matrix @ AbstractLiteral::Matrix(_, _)) => {
769                matrix.unwrap_list().cloned()
770            }
771            Expression::Atomic(
772                _,
773                Atom::Literal(Literal::AbstractLiteral(matrix @ AbstractLiteral::Matrix(_, _))),
774            ) => matrix.unwrap_list().map(|elems| {
775                elems
776                    .clone()
777                    .into_iter()
778                    .map(|x: Literal| Expression::Atomic(Metadata::new(), Atom::Literal(x)))
779                    .collect_vec()
780            }),
781            _ => None,
782        }
783    }
784
785    /// If the expression is a matrix, gets it elements and index domain.
786    ///
787    /// **Consider using the safer [`Expression::unwrap_list`] instead.**
788    ///
789    /// It is generally undefined to edit the length of a matrix unless it is a list (as defined by
790    /// [`Expression::unwrap_list`]). Users of this function should ensure that, if the matrix is
791    /// reconstructed, the index domain and the number of elements in the matrix remain the same.
792    pub fn unwrap_matrix_unchecked(self) -> Option<(Vec<Expression>, Domain)> {
793        match self {
794            Expression::AbstractLiteral(_, AbstractLiteral::Matrix(elems, domain)) => {
795                Some((elems.clone(), domain))
796            }
797            Expression::Atomic(
798                _,
799                Atom::Literal(Literal::AbstractLiteral(AbstractLiteral::Matrix(elems, domain))),
800            ) => Some((
801                elems
802                    .clone()
803                    .into_iter()
804                    .map(|x: Literal| Expression::Atomic(Metadata::new(), Atom::Literal(x)))
805                    .collect_vec(),
806                domain,
807            )),
808
809            _ => None,
810        }
811    }

813    /// For a Root expression, extends the inner vec with the given vec.
814    ///
815    /// # Panics
816    /// Panics if the expression is not Root.
817    pub fn extend_root(self, exprs: Vec<Expression>) -> Expression {
818        match self {
819            Expression::Root(meta, mut children) => {
820                children.extend(exprs);
821                Expression::Root(meta, children)
822            }
823            _ => panic!("extend_root called on a non-Root expression"),
824        }
825    }

827    /// Converts the expression to a literal, if possible.
828    pub fn to_literal(self) -> Option<Literal> {
829        match self {
830            Expression::Atomic(_, Atom::Literal(lit)) => Some(lit),
831            Expression::AbstractLiteral(_, abslit) => {
832                Some(Literal::AbstractLiteral(abslit.clone().as_literals()?))
833            }
834            Expression::Neg(_, e) => {
835                let Literal::Int(i) = e.to_literal()? else {
836                    bug!("negated literal should be an int");
837                };
838
839                Some(Literal::Int(-i))
840            }
841
842            _ => None,
843        }
844    }

877impl Display for Expression {
878    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
879        match &self {
880            Expression::AbstractLiteral(_, l) => l.fmt(f),
881            Expression::Comprehension(_, c) => c.fmt(f),
882            Expression::UnsafeIndex(_, e1, e2) | Expression::SafeIndex(_, e1, e2) => {
883                write!(f, "{e1}{}", pretty_vec(e2))
884            }
885            Expression::UnsafeSlice(_, e1, es) | Expression::SafeSlice(_, e1, es) => {
886                let args = es
887                    .iter()
888                    .map(|x| match x {
889                        Some(x) => format!("{}", x),
890                        None => "..".into(),
891                    })
892                    .join(",");
893
894                write!(f, "{e1}[{args}]")
895            }
896            Expression::InDomain(_, e, domain) => {
897                write!(f, "__inDomain({e},{domain})")
898            }
899            Expression::Root(_, exprs) => {
900                write!(f, "{}", pretty_expressions_as_top_level(exprs))
901            }
902            Expression::DominanceRelation(_, expr) => write!(f, "DominanceRelation({})", expr),
903            Expression::FromSolution(_, expr) => write!(f, "FromSolution({})", expr),
904            Expression::Atomic(_, atom) => atom.fmt(f),
905            Expression::Scope(_, submodel) => write!(f, "{{\n{submodel}\n}}"),
906            Expression::Abs(_, a) => write!(f, "|{}|", a),
907            Expression::Sum(_, e) => {
908                write!(f, "Sum({e})")
909            }
910            Expression::Product(_, expressions) => {
911                write!(f, "Product({})", pretty_vec(expressions))
912            }
913            Expression::Min(_, e) => {
914                write!(f, "min({e})")
915            }
916            Expression::Max(_, e) => {
917                write!(f, "max({e})")
918            }
919            Expression::Not(_, expr_box) => {
920                write!(f, "Not({})", expr_box.clone())
921            }
922            Expression::Or(_, e) => {
923                write!(f, "or({e})")
924            }
925            Expression::And(_, e) => {
926                write!(f, "and({e})")
927            }
928            Expression::Imply(_, box1, box2) => {
929                write!(f, "({}) -> ({})", box1, box2)
930            }
931            Expression::Eq(_, box1, box2) => {
932                write!(f, "({} = {})", box1.clone(), box2.clone())
933            }
934            Expression::Neq(_, box1, box2) => {
935                write!(f, "({} != {})", box1.clone(), box2.clone())
936            }
937            Expression::Geq(_, box1, box2) => {
938                write!(f, "({} >= {})", box1.clone(), box2.clone())
939            }
940            Expression::Leq(_, box1, box2) => {
941                write!(f, "({} <= {})", box1.clone(), box2.clone())
942            }
943            Expression::Gt(_, box1, box2) => {
944                write!(f, "({} > {})", box1.clone(), box2.clone())
945            }
946            Expression::Lt(_, box1, box2) => {
947                write!(f, "({} < {})", box1.clone(), box2.clone())
948            }
949            Expression::FlatSumGeq(_, box1, box2) => {
950                write!(f, "SumGeq({}, {})", pretty_vec(box1), box2.clone())
951            }
952            Expression::FlatSumLeq(_, box1, box2) => {
953                write!(f, "SumLeq({}, {})", pretty_vec(box1), box2.clone())
954            }
955            Expression::FlatIneq(_, box1, box2, box3) => write!(
956                f,
957                "Ineq({}, {}, {})",
958                box1.clone(),
959                box2.clone(),
960                box3.clone()
961            ),
962            Expression::AllDiff(_, e) => {
963                write!(f, "allDiff({e})")
964            }
965            Expression::Bubble(_, box1, box2) => {
966                write!(f, "{{{} @ {}}}", box1.clone(), box2.clone())
967            }
968            Expression::SafeDiv(_, box1, box2) => {
969                write!(f, "SafeDiv({}, {})", box1.clone(), box2.clone())
970            }
971            Expression::UnsafeDiv(_, box1, box2) => {
972                write!(f, "UnsafeDiv({}, {})", box1.clone(), box2.clone())
973            }
974            Expression::UnsafePow(_, box1, box2) => {
975                write!(f, "UnsafePow({}, {})", box1.clone(), box2.clone())
976            }
977            Expression::SafePow(_, box1, box2) => {
978                write!(f, "SafePow({}, {})", box1.clone(), box2.clone())
979            }
980            Expression::MinionDivEqUndefZero(_, box1, box2, box3) => {
981                write!(
982                    f,
983                    "DivEq({}, {}, {})",
984                    box1.clone(),
985                    box2.clone(),
986                    box3.clone()
987                )
988            }
989            Expression::MinionModuloEqUndefZero(_, box1, box2, box3) => {
990                write!(
991                    f,

993                    box1.clone(),
994                    box2.clone(),
995                    box3.clone()
996                )
997            }
998            Expression::FlatWatchedLiteral(_, x, l) => {
999                write!(f, "WatchedLiteral({},{})", x, l)
1000            }
1001            Expression::MinionReify(_, box1, box2) => {
1002                write!(f, "Reify({}, {})", box1.clone(), box2.clone())
1003            }
1004            Expression::MinionReifyImply(_, box1, box2) => {
1005                write!(f, "ReifyImply({}, {})", box1.clone(), box2.clone())
1006            }
1007            Expression::MinionWInIntervalSet(_, atom, intervals) => {
1008                let intervals = intervals.iter().join(",");
1009                write!(f, "__minion_w_inintervalset({atom},{intervals})")
1010            }
1011            Expression::AuxDeclaration(_, n, e) => {
1012                write!(f, "{} =aux {}", n, e.clone())
1013            }
1014            Expression::UnsafeMod(_, a, b) => {
1015                write!(f, "{} % {}", a.clone(), b.clone())
1016            }

1018                write!(f, "SafeMod({},{})", a.clone(), b.clone())
1019            }
1020            Expression::Neg(_, a) => {
1021                write!(f, "-({})", a.clone())
1022            }
1023            Expression::Minus(_, a, b) => {
1024                write!(f, "({} - {})", a.clone(), b.clone())
1025            }
1026            Expression::FlatAllDiff(_, es) => {
1027                write!(f, "__flat_alldiff({})", pretty_vec(es))
1028            }
1029            Expression::FlatAbsEq(_, a, b) => {
1030                write!(f, "AbsEq({},{})", a.clone(), b.clone())
1031            }
1032            Expression::FlatMinusEq(_, a, b) => {
1033                write!(f, "MinusEq({},{})", a.clone(), b.clone())
1034            }
1035            Expression::FlatProductEq(_, a, b, c) => {
1036                write!(
1037                    f,
1038                    "FlatProductEq({},{},{})",
1039                    a.clone(),
1040                    b.clone(),
1041                    c.clone()
1042                )
1043            }
1044            Expression::FlatWeightedSumLeq(_, cs, vs, total) => {
1045                write!(
1046                    f,
1047                    "FlatWeightedSumLeq({},{},{})",
1048                    pretty_vec(cs),
1049                    pretty_vec(vs),
1050                    total.clone()
1051                )
1052            }
1053            Expression::FlatWeightedSumGeq(_, cs, vs, total) => {

1058                    pretty_vec(vs),
1059                    total.clone()
1060                )
1061            }
1062            Expression::MinionPow(_, atom, atom1, atom2) => {
1063                write!(f, "MinionPow({},{},{})", atom, atom1, atom2)
1064            }
1065            Expression::MinionElementOne(_, atoms, atom, atom1) => {
1066                let atoms = atoms.iter().join(",");
1067                write!(f, "__minion_element_one([{atoms}],{atom},{atom1})")
1068            }
1069        }
1070    }
